CREATE PROCEDURE [dbo].[AddNote]
    @UserId int,
	@Title varchar(50),  
    @Body varchar(MAX),
	@CreatedDate Datetime
AS  
BEGIN  
	INSERT INTO [dbo].[Notes](
		[UserId],
		[Title],
		[Body],
		[CreationDate]) 
	VALUES(
		@UserId,
		@Title,
		@Body,
		@CreatedDate)  
END  
GO
