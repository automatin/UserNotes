CREATE PROCEDURE[dbo].[GetNoteById]  
@NoteId int
AS  
BEGIN  
select 
	[Id],
	[Title],
	[Body],
	[CreationDate],
	[UserId] 
from [dbo].[Notes] where Id = @NoteId;  
END  
GO
