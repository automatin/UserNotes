CREATE PROCEDURE [dbo].[UpdateNote]
	@NoteId int,
    @UserId int,
	@Title varchar(50),  
    @Body varchar(MAX),
	@CreatedDate Datetime
AS  
BEGIN  
	UPDATE [dbo].[Notes] SET
		[UserId] = @UserId,
		[Title] = @Title,
		[Body] = @Body,
		[CreationDate] = @CreatedDate 
	WHERE[dbo].[Notes].Id = @NoteId
		 
END  
GO
